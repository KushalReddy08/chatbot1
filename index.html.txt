    <script>
        const chatMessages = document.getElementById('chatMessages');
        const userInput = document.getElementById('userInput');
        const sendBtn = document.getElementById('sendBtn');

        function createMsgRow(text, sender) {
            const row = document.createElement('div');
            row.className = 'message-row ' + sender;

            const meta = document.createElement('div');
            meta.className = sender === 'user' ? 'user-label' : 'bot-label';
            meta.innerText = sender === 'user' ? 'You' : 'Kushal';

            const bubble = document.createElement('div');
            bubble.className = 'bubble';
            bubble.innerText = text;

            if(sender === 'user') {
                row.appendChild(bubble);
                row.appendChild(meta);
            } else {
                row.appendChild(meta);
                row.appendChild(bubble);
            }
            chatMessages.appendChild(row);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        function appendTyping() {
            const typing = document.createElement('div');
            typing.className = 'typing';
            typing.id = 'typingIndicator';
            typing.innerText = 'Kushal is typing...';
            chatMessages.appendChild(typing);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        function removeTyping() {
            const typing = document.getElementById('typingIndicator');
            if (typing) typing.remove();
        }

        function sendMessage() {
            const message = userInput.value.trim();
            if (!message) return;
            createMsgRow(message, 'user');
            userInput.value = '';
            appendTyping();
            fetch('/chat', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ message })
            })
            .then((response) => response.json())
            .then((data) => {
                removeTyping();
                createMsgRow(data.reply, 'bot');
            })
            .catch(() => {
                removeTyping();
                createMsgRow('Sorry, there was a connection error.', 'bot');
            });
        }

        sendBtn.addEventListener('click', sendMessage);
        userInput.addEventListener('keydown', function(event) {
            if (event.key === 'Enter') sendMessage();
        });

        window.onload = () => {
            userInput.focus();
            createMsgRow("Hello! ðŸ‘‹ I'm Kushal, your AI assistant. How can I help you today?", 'bot');
        };
    </script>
